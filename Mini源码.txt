local Library = {}
Library.Flags = {}

Library.AutoSize = true

local function GetOptimalSize()
	local screenSize = workspace.CurrentCamera.ViewportSize
	local width = math.clamp(screenSize.X * 0.75, 550, 950)
	local height = math.clamp(screenSize.Y * 0.85, 400, 650)
	return width, height
end

function Library:Window(title)
	TitleText.Text = title or "MiniUI"
	return self
end

Library.new = Library.Window
Library.New = Library.Window

function Library:SetAutoSize(enabled)
	Library.AutoSize = enabled
	if enabled and Library.IsOpen then
		local width, height = GetOptimalSize()
		TweenService:Create(Main, TweenInfo.new(0.3), {
			Size = UDim2.new(0, width, 0, height)
		}):Play()
	end
end

function Library:Toggle()
	Library.IsOpen = not Library.IsOpen
	
	if Library.IsOpen then
		Main.Visible = true
		
		if Library.AutoSize then
			local width, height = GetOptimalSize()
			TweenService:Create(Main, TweenInfo.new(0.5, Enum.EasingStyle.Back), {
				Size = UDim2.new(0, width, 0, height)
			}):Play()
		else
			TweenService:Create(Main, TweenInfo.new(0.5, Enum.EasingStyle.Back), {
				Size = UDim2.new(0, 700, 0, 500)
			}):Play()
		end
		
		TweenService:Create(ToggleButton, TweenInfo.new(0.3), {
			BackgroundColor3 = Color3.fromRGB(189, 147, 249),  -- Dracula: #BD93F9
			ImageColor3 = Color3.fromRGB(255, 255, 255),
			Rotation = 180
		}):Play()
	else
		TweenService:Create(Main, TweenInfo.new(0.3, Enum.EasingStyle.Back, Enum.EasingDirection.In), {
			Size = UDim2.new(0, 0, 0, 0)
		}):Play()
		
		TweenService:Create(ToggleButton, TweenInfo.new(0.3), {
			BackgroundColor3 = Color3.fromRGB(40, 42, 54),      -- Dracula: #282A36
			ImageColor3 = Color3.fromRGB(255, 121, 198),        -- Dracula: #FF79C6
			Rotation = 0
		}):Play()
		
		task.wait(0.3)
		Main.Visible = false
	end
end

local searchConnection = SearchBox:GetPropertyChangedSignal("Text"):Connect(function()
	local searchText = string.lower(SearchBox.Text)
	for _, elementData in pairs(Library.AllElements) do
		if elementData.Type and elementData.Object then
			local text = ""
			if elementData.Type == "Button" then
				text = string.lower(elementData.Object.Text)
			elseif elementData.Type == "Label" then
				text = string.lower(elementData.Text or "")
			elseif elementData.Type == "Toggle" then
				text = string.lower(elementData.Object:FindFirstChild("Label") and elementData.Object.Label.Text or "")
			elseif elementData.Type == "Textbox" then
				text = string.lower(elementData.Object:FindFirstChild("Label") and elementData.Object.Label.Text or "")
			elseif elementData.Type == "Slider" then
				text = string.lower(elementData.Object:FindFirstChild("Label") and elementData.Object.Label.Text or "")
			elseif elementData.Type == "Dropdown" then
				text = string.lower(elementData.Text or "")
			end
			
			if searchText == "" or string.find(text, searchText) then
				elementData.Object.Visible = true
				if elementData.Parent then
					elementData.Parent.Visible = true
				end
			else
				elementData.Object.Visible = false
			end
		end
	end
	
	for _, tab in pairs(Library.Tabs) do
		for _, section in pairs(tab.Sections) do
			local hasVisible = false
			for _, child in pairs(section.Frame:GetDescendants()) do
				if child:IsA("TextButton") or child:IsA("TextLabel") or child:IsA("TextBox") or child:IsA("Frame") then
					if child.Visible and child ~= section.Frame and child ~= section.Header then
						hasVisible = true
						break
					end
				end
			end
			if not hasVisible and searchText ~= "" then
				section.Frame.Visible = false
			else
				section.Frame.Visible = true
			end
		end
	end
end)
table.insert(UISConnections, searchConnection)

function Library:Destroy()
	for _, conn in pairs(UISConnections) do
		if conn then
			pcall(function() conn:Disconnect() end)
		end
	end
	UISConnections = {}
	
	if MiniUI then
		MiniUI:Destroy()
	end
	
	for k, v in pairs(Library) do
		Library[k] = nil
	end
	Library = nil
end

				if not precise then
					val = math.floor(val)

				SliderFill.Size = UDim2.new(pos, 0, 1, 0)
				ValueLabel.Text = tostring(value)
				Library.Flags[flag] = value
				if callback then callback(value) end

					TweenService:Create(OptionBtn, TweenInfo.new(0.15), {BackgroundColor3 = Color3.fromRGB(68, 71, 90)}):Play()
				end)
				
				OptionBtn.MouseButton1Click:Connect(function()
					selected = option
					Library.Flags[flag] = selected
					DropdownButton.Text = "  " .. text .. ": " .. selected .. " ▼"
					if expanded then
						ToggleDropdown()
					end
					if callback then callback(selected) end
				end)
			end
		
			for _, option in ipairs(currentOptions) do
				CreateOptionButton(option)
			end
			
			DropdownButton.MouseButton1Click:Connect(ToggleDropdown)
			
			table.insert(Library.AllElements, {
				Type = "Dropdown",
				Object = DropdownFrame,
				Text = text,
				Parent = ElementContainer
			})
			
			local DropdownAPI = {}
			
			function DropdownAPI:SetValue(val)
				if table.find(currentOptions, val) then
					selected = val
					DropdownButton.Text = "  " .. text .. ": " .. selected .. " ▼"
					Library.Flags[flag] = selected
					if callback then callback(selected) end
				end
			end
			
			function DropdownAPI:GetValue()
				return selected
			end
			
			function DropdownAPI:SetOptions(newOptions)
				currentOptions = newOptions or {}
				
				for _, child in ipairs(OptionContainer:GetChildren()) do
					if child:IsA("TextButton") then 
						child:Destroy() 
					end
				end
				
				for _, option in ipairs(currentOptions) do
					CreateOptionButton(option)
				end
				
				if expanded then
					local totalHeight = GetTotalHeight()
					local targetHeight = math.min(42 + totalHeight, 250)
					DropdownFrame.Size = UDim2.new(1, 0, 0, targetHeight)
				end
				
				if selected and not table.find(currentOptions, selected) then
					selected = nil
					Library.Flags[flag] = nil
					DropdownButton.Text = "  " .. text .. " ▼"
				end
			end
			
			DropdownAPI.Refresh = DropdownAPI.SetOptions
			
			return DropdownAPI
		end
		
		table.insert(Tab.Sections, Section)
		return Section
	end
	
	table.insert(Library.Tabs, Tab)
	return Tab
end

task.spawn(function()
	task.wait(0.5)
	if Library then
		Library:Toggle()
	end
end)

return Library
